        def encontrar  
        
                valor = "["

                0.upto(matriz.size - 1) do |i|

                        (matriz[i].size).times do |j| 
                                
                               if (yield(matriz[i][j])) 
                               
                                        valor += "[#{i},#{j}]"

                              end

                        end
                end

                valor += "]"

                valor

        end
