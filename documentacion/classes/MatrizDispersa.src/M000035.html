        def to_f

                flotante = Array.new(matriz.size - 1)
                for i in 0...matriz.size
                        #=== Hay datos en la fila
                        if matriz[i] != nil
                                
                                flotante[i] = Hash.new()
                                matriz[i].each do |key, value|
                                        flotante[i][key] = matriz[i][key].to_f
                                end
                        
                        end

                end
                MatrizDispersa.new(flotante)

        end
