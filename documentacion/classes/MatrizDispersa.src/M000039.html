        def max

                maximo = 0.to_f
                for i in 0...matriz.size
                        #=== Hay datos en la fila
                        if matriz[i] != nil
                                matriz[i].each do |key, value|
                                        if matriz[i][key].to_f > maximo
                                                maximo = matriz[i][key].to_f
                                        end
                                end                        
                        end
                end
                maximo

        end
